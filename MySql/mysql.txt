SELECT MONTHNAME(date_col) AS month, DAY(LAST_DAY(date_col)) AS days_in_month
FROM (
  SELECT DATE_ADD(DATE(DATE_FORMAT(NOW() ,'%Y-01-01')), INTERVAL n.n MONTH) AS date_col
  FROM (
    SELECT 0 n UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL 
    SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL
    SELECT 8 UNION ALL SELECT 9 UNION ALL SELECT 10 UNION ALL SELECT 11
  ) n
) AS months;
